    ```gdscript
    # Sub-test (b) — clip_polygon_to_subrect determinism.
    # Calls clip_polygon_to_subrect 10 times with identical inputs; asserts output hash is identical.
    func _subtest_clip_polygon_determinism() -> void:
        # Fixed test polygon: unit square with TL at origin
        var test_polygon := PackedVector2Array([
            Vector2(0, 0),
            Vector2(1, 0),
            Vector2(1, 1),
            Vector2(0, 1),
        ])
        # Fixed sub_rect: centered 0.5×0.5 sub-rect (origin (0.25, 0.25), size 0.5×0.5)
        var test_sub_rect := Rect2(0.25, 0.25, 0.5, 0.5)
        var test_full_tile_size := Vector2(1.0, 1.0)
        var hashes: Array[int] = []
        for i in range(10):
            var clipped: PackedVector2Array = PentaTileSynthesis.clip_polygon_to_subrect(
                test_polygon, test_sub_rect, test_full_tile_size
            )
            hashes.append(clipped.hash())
        var first_hash := hashes[0]
        for i in range(1, 10):
            assert(hashes[i] == first_hash,
                "clip_polygon_to_subrect non-deterministic on run %d: %d != %d" % [i, hashes[i], first_hash])
        print("Sub-test (b) — clip_polygon_to_subrect determinism: PASS (10 invocations, hash=%d)" % first_hash)
    ```
