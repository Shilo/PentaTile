    **Implement _build_terrain_index() (per D-04, D-15, and Phase 9 blueprint §4.4):**

    Add after `_resolve_source_id()` method. This is a transient index — never persisted, rebuilt on every terrain_group set:

    ```gdscript
    # Transient terrain index — maps terrain_id (0-indexed) to TerrainEntry.
    # Built by _build_terrain_index(); cleared when terrain_group is null.
    # TerrainEntry is a lightweight inner class (not a Resource) since the index
    # is never persisted.
    var _terrain_index: Dictionary = {}

    func _build_terrain_index() -> void:
    	_terrain_index.clear()
    	if terrain_group == null or tile_set == null:
    		return

    	for terrain_id in range(terrain_group.layouts.size()):
    		var layout := terrain_group.layouts[terrain_id]
    		if layout == null:
    			continue
    		var tiles: Array[Vector2i] = []
    		var source_count := tile_set.get_source_count()
    		for src_idx in range(source_count):
    			var source := tile_set.get_source(src_idx) as TileSetAtlasSource
    			if source == null:
    				continue
    			var tiles_count := source.get_tiles_count()
    			for t in range(tiles_count):
    				var coord := source.get_tile_id(t)
    				var alt_count := source.get_alternative_tiles_count(coord)
    				# Per GAP-01: scan ALL alternative tiles, not just alt_id=0
    				for alt_id in range(alt_count):
    					var tile_data := source.get_tile_data(coord, alt_id)
    					if tile_data == null:
    						continue
    					var td_terrain := tile_data.terrain
    					# Per GAP-02: skip tiles without center bit
    					if td_terrain < 0:
    						continue
    					if td_terrain == terrain_id:
    						tiles.append(coord)
    						break  # coord already indexed, skip remaining alts
    		_terrain_index[terrain_id] = {"layout": layout, "tiles": tiles}

    # Emits when terrain_group properties change. Follows _on_layout_changed pattern.
    func _on_terrain_group_changed() -> void:
    	_build_terrain_index()
    	_queue_rebuild()
    ```

    **Key properties of this implementation:**
    - `_terrain_index` is a `Dictionary[int, Dictionary]` — not a Resource, not persisted
    - `terrain_mode()` is NOT called during index building (only terrain metadata is read). terrain_mode() is consumed by the peering-bits-to-mask converter in variation discovery (Plan 03)
    - The `tiles` array stores atlas coords — not TileData references (TileData is a transient object; coords are stable)
    - `break` after append avoids duplicate entries when multiple alternative tiles of the same coord share the same terrain

    **1. Implement `_resolve_terrain_id(cell: Vector2i) -> int` (per D-04 resolution order):**

    Add after `_build_terrain_index()`:

    ```gdscript
    ## Resolve which terrain_id a logic cell belongs to.
    ## Resolution order (D-04):
    ##   1. Read penta_terrain_id from logic cell's custom data — if >= 0, use that
    ##   2. Read TileData.terrain from the cell's base tile (alt_id=0) — if >= 0, use that
    ##   3. Default to 0 (first terrain in terrain_group.layouts)
    ## Returns 0 when terrain_group is null or cell is empty.
    func _resolve_terrain_id(cell: Vector2i) -> int:
    	if terrain_group == null:
    		return 0
    	var source_id := get_cell_source_id(cell)
    	if source_id == -1:
    		return 0
    	# Step 1: custom data layer override (D-03)
    	var custom := get_cell_tile_data(cell)
    	if custom != null:
    		var penta_terrain := custom.get_custom_data("penta_terrain_id")
    		if typeof(penta_terrain) == TYPE_INT and penta_terrain >= 0:
    			return penta_terrain
    	# Step 2: Godot native terrain property (D-03)
    	var source := tile_set.get_source(source_id) as TileSetAtlasSource
    	if source != null:
    		var atlas_coords := get_cell_atlas_coords(cell)
    		if source.has_tile(atlas_coords):
    			var tile_data := source.get_tile_data(atlas_coords, 0)
    			if tile_data != null and tile_data.terrain >= 0:
    				return tile_data.terrain
    	# Step 3: default terrain (D-04)
    	return 0
    ```
