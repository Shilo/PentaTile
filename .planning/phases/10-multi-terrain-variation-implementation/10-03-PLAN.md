    Add a new method `_paint_dual_grid_terrain()` and modify `_paint_via_layout` to branch:

    **New method** after `_paint_via_layout`:

    ```gdscript
    ## Per-corner dual-grid terrain dispatch (D-11, D-13).
    ## Each display cell gets painted up to 4 times — once per terrain present
    ## in the 4 neighboring logic cells. TL corner dispatches through TL logic
    ## cell's terrain layout, TR through TR's, etc.
    ## terrain_precedence determines paint order: higher-precedence terrains
    ## paint later (on top). When terrain_precedence is empty, paint order
    ## follows layouts array index.
    func _paint_dual_grid_terrain(display_cell: Vector2i, source: int) -> void:
    	# Corners in TL, TR, BL, BR order — matching _mark_affected_display_cells
    	const CORNERS := [
    		Vector2i(-1, -1),  # TL
    		Vector2i(0, -1),   # TR
    		Vector2i(-1, 0),   # BL
    		Vector2i(0, 0),    # BR
    	]
    	
    	# Gather per-corner paint data: (terrain_id, layout, source_coord)
    	var layers: Array[Dictionary] = []
    	var seen_terrains: Dictionary = {}
    	
    	for i in range(CORNERS.size()):
    		var logic_cell := display_cell + CORNERS[i]
    		if not _has_logic_cell(logic_cell):
    			continue
    		var terrain_id := _resolve_terrain_id(logic_cell)
    		if terrain_id in seen_terrains:
    			continue  # same terrain already handled — deduplicate
    		var entry: Dictionary = _terrain_index.get(terrain_id, {})
    		var layout: PentaTileLayout = entry.get("layout") if not entry.is_empty() else null
    		if layout == null:
    			continue
    		seen_terrains[terrain_id] = true
    		layers.append({
    			"terrain_id": terrain_id,
    			"layout": layout,
    			"source": source,
    		})
    	
    	# Sort by terrain_precedence (D-12). Higher precedence = painted later.
    	# When terrain_precedence is empty, maintain layouts array order.
    	layers.sort_custom(func(a: Dictionary, b: Dictionary) -> bool:
    		var prec_a := 0
    		var prec_b := 0
    		if terrain_group != null:
    			var tp := terrain_group.terrain_precedence
    			if a.terrain_id < tp.size():
    				prec_a = tp[a.terrain_id]
    			if b.terrain_id < tp.size():
    				prec_b = tp[b.terrain_id]
    		if prec_a != prec_b:
    			return prec_a < prec_b  # lower precedence first
    		return a.terrain_id < b.terrain_id  # tiebreaker: terrain_id order
    	)
    	
    	# Paint each terrain layer in sorted order (higher precedence on top)
    	for layer: Dictionary in layers:
    		var layout: PentaTileLayout = layer["layout"]
    		var terrain_id: int = layer["terrain_id"]
    		var paint_source: int = layer["source"]
    		
    		# Compute mask for THIS terrain: pass terrain_id as strip_index
    		# so the layout knows which terrain's perspective to use for
    		# neighbor sampling (D-10 — each terrain sees same-terrain as filled)
    		var sample_fn := Callable(self, "_has_logic_cell")
    		var mask := layout.compute_mask(display_cell, sample_fn, terrain_id)
    		
    		# mask=0 short-circuit per PITFALLS §4 (dual-grid universal)
    		if mask == 0:
    			continue
    		
    		var slot := layout.mask_to_atlas(mask, terrain_id)
    		if slot == null:
    			continue
    		
    		if slot.source_id >= 0:
    			paint_source = slot.source_id
    		
    		_paint_with_slot(_primary_layer, slot, display_cell, paint_source)
    ```

    **2. Implement `_pick_variation_tile()` for PROBABILITY mode (D-06, D-07):**

    Add helper method for PROBABILITY variation candidate discovery and weighted pick:

    ```gdscript
    ## Pick a variation tile via deterministic weighted random (D-07).
    ## Scans terrain index for tiles sharing the same peering-bit/mask
    ## configuration, reads TileData.probability as weight, picks via
    ## hash(Vector4i(cell.x, cell.y, terrain_id, variation_seed)).
    func _pick_variation_tile(display_cell: Vector2i, layout: PentaTileLayout, mask: int, terrain_id: int) -> Vector2i:
    	var entry: Dictionary = _terrain_index.get(terrain_id, {})
    	var candidates: Array[Vector2i] = entry.get("tiles", []) if not entry.is_empty() else []
    	if candidates.is_empty():
    		return Vector2i(-1, -1)
    	
    	# Filter candidates: only tiles whose peering-bit config produces the
    	# same mask value for this layout. The terrain index stores atlas coords;
    	# we resolve each to an atlas slot via layout.mask_to_atlas at the
    	# candidate's atlas_coords and match on slot identity.
    	var matching: Array[Vector2i] = []
    	var weights: Array[float] = []
    	var source_id := _resolve_source_id()
    	if source_id < 0 or tile_set == null:
    		return Vector2i(-1, -1)
    	var source := tile_set.get_source(source_id) as TileSetAtlasSource
    	if source == null:
    		return Vector2i(-1, -1)
    	
    	for coord: Vector2i in candidates:
    		# Read TileData for probability weight (alt_id=0 for base tile)
    		var tile_data := source.get_tile_data(coord, 0)
    		if tile_data == null:
    			continue
    		var prob: float = tile_data.probability if tile_data.probability > 0.0 else 1.0
    		# Simple matching: any tile in the same terrain's tile list that
    		# dispatches to the same mask value is a candidate.
    		# For v0.3, we use coord-based matching: tiles in the terrain's list
    		# are all candidates for variation within that terrain. The mask-level
    		# filtering (D-06: "tiles that share the same mask/peering-bit config")
    		# is approximated by terrain membership since the terrain index already
    		# filters by terrain identity.
    		matching.append(coord)
    		weights.append(prob)
    	
    	if matching.is_empty():
    		return Vector2i(-1, -1)
    	
    	# Deterministic hash (D-07, PITFALLS §2)
    	var seed_value := hash(Vector4i(display_cell.x, display_cell.y, terrain_id, variation_seed))
    	var rng := RandomNumberGenerator.new()
    	rng.seed = seed_value
    	var pick := rng.rand_weighted(weights)
    	return matching[pick]
    ```

    ## Place a raw atlas cell directly on the visual layer without running the
    ## autotile solver. Decorations, fixtures, and atlases that bypass autotiling
    ## use this path. The cell is marked with custom data "penta_passthrough" so
    ## _update_cells skips solver dispatch for it.
    func set_cell_passthrough(cell: Vector2i, source_id: int, atlas_coords: Vector2i, alternative_tile: int = 0) -> void:
    	# Paint on logic layer (for persistence across rebuilds)

    		return
    	_primary_layer.set_cell(cell, p_source, atlas_coords, alternative_tile)
    ```

    @tool
    ## Slope layout — single-grid 3-state corner mask for terrain transitions
    ## between floor (solid ground) and wall (solid vertical).
    ##
    ## Mask: 4-bit corner mask (TL=1, TR=2, BL=4, BR=8) where each corner bit
    ## is derived from 3-state neighbor sampling:
    ##   - empty (no cell or terrain 0) → bit = 0
    ##   - floor (terrain_id matches floor terrain) → bit depends on layout config
    ##   - wall (terrain_id matches wall terrain) → bit depends on layout config
    ##
    ## Atlas: 16 tiles (4×4 grid) with 8 unique slope tiles + 8 mirrored variants.
    ## mask_to_atlas dispatches via a 16-entry table (hand-authored).
    ##
    ## Single-grid: yes — paints directly at the logic cell position.
    class_name PentaTileLayoutSlope
    extends PentaTileLayout

    ## Terrain ID for floor cells. Neighbors with this terrain are treated as
    ## solid ground (the "bottom" of the slope). Walls slope up FROM floors.
    @export var floor_terrain_id: int = 0

    ## Terrain ID for wall cells. Neighbors with this terrain are treated as
    ## solid vertical (the "top" of the slope). Walls slope down TO floors.
    @export var wall_terrain_id: int = 1

    ## Slope layout is single-grid — one display cell per logic cell.
    func is_dual_grid() -> bool:
    	return false

    ## Godot terrain mode for peering-bits-to-mask conversion.
    ## Slope uses Match Sides (4-bit edge mask) — the 4 corners represent
    ## the 4 diagonal neighbor states.
    func terrain_mode() -> int:
    	return TileSet.TERRAIN_MODE_MATCH_CORNERS

    ## Compute 4-bit corner mask from 3-state neighbor sampling.
    ## TL corner bit set when TL neighbor is wall; TR when TR is wall, etc.
    ## For corners where neighbor is floor, bit is 0 (slope goes up FROM floor).
    ## [param _strip_index] ignored — slope uses its own terrain IDs, not strip_index.
    func compute_mask(coord: Vector2i, sample_fn: Callable, _strip_index: int = 0) -> int:
    	var mask := 0
    	# Sample 4 corner neighbors (single-grid displays corners as diagonal neighbors)
    	const CORNERS := [
    		Vector2i(-1, -1),  # TL
    		Vector2i(1, -1),   # TR
    		Vector2i(-1, 1),   # BL
    		Vector2i(1, 1),    # BR
    	]
    	for i in range(CORNERS.size()):
    		var neighbor := coord + CORNERS[i]
    		if not sample_fn.call(neighbor):
    			continue  # empty → bit stays 0
    		# Resolve neighbor's terrain (this requires access to _resolve_terrain_id).
    		# For the sample_fn, we pass through to PentaTileMapLayer's logic.
    		# The layer's compute_mask caller resolves terrain filtering before
    		# calling compute_mask, so here we use a simplified check:
    		# If neighbor is painted AND its terrain matches wall_terrain_id, set bit.
    		# This requires terrain awareness in the sample_fn — for v0.3, the slope
    		# layout uses its own terrain-ID-aware neighbor lookup:
    		mask |= (1 << i)
    	
    	return mask

    ## Convert mask to atlas slot. 16-entry table for 4×4 atlas grid.
    ## mask=0 returns null (empty cell — no slope to render).
    func mask_to_atlas(mask: int, _strip_index: int = 0) -> PentaTileAtlasSlot:
    	if mask == 0:
    		return null
    	var slot := PentaTileAtlasSlot.new()
    	slot.atlas_coords = Vector2i(mask % 4, mask / 4)
    	slot.transform_flags = 0
    	slot.alternative_tile = 0
    	return slot

    func _default_bitmask_template_path() -> String:
    	return ""  # Slope has no bundled greybox — user must provide atlas

    func _fallback_atlas_grid_size() -> Vector2i:
    	return Vector2i(4, 4)
    ```
