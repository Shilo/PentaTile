```gdscript
func _mask_at(display_cell: Vector2i) -> int:
    var mask := 0
    if _has_logic_cell(display_cell + _TL):
        mask |= 1
    if _has_logic_cell(display_cell + _TR):
        mask |= 2
    if _has_logic_cell(display_cell + _BL):
        mask |= 4
    if _has_logic_cell(display_cell + _BR):
        mask |= 8
    return mask
```
